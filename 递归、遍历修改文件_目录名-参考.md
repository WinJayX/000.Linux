数据参考
for i in `find . -depth`; do mv $i `echo $i |tr [A-Z] [a-z]` ; done

------只对当前目录有效。。。



find . -depth -exec rename -n -e 's/A-Z/a-z/g' {} \;


find <dir> -depth -exec rename 'y/a-z/A-Z/' {} \;

find /opt/ftpfile/logs --depth | \
while read name; do \
na=$(echo $name | tr '[A-Z]' '[a-z]') \
if [[ $name != $na ]]; then \
mv "$name" $na \
fi \
done
